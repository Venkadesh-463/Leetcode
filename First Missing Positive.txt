class Solution:
    def firstMissingPositive(self, nums: list[int]) -> int:
        n = len(nums)
        
        # 1. Place each number in its target index: nums[i] should be at index nums[i] - 1
        for i in range(n):
            # While the current number is in the valid range [1, n] 
            # and is NOT already at its correct position:
            while 1 <= nums[i] <= n and nums[nums[i] - 1] != nums[i]:
                # Swap it to its target position
                target_idx = nums[i] - 1
                nums[i], nums[target_idx] = nums[target_idx], nums[i]
        
        # 2. Scan the array to find the first index that doesn't match its value
        for i in range(n):
            if nums[i] != i + 1:
                return i + 1
        
        # 3. If all numbers from 1 to n are present, the answer is n + 1
        return n + 1
