import math

class Solution:
    def getPermutation(self, n: int, k: int) -> str:
        # 1. Create a list of numbers to use: [1, 2, 3, ..., n]
        numbers = [str(i) for i in range(1, n + 1)]
        
        # 2. Adjust k to be 0-indexed for easier math
        k -= 1
        
        res = []
        # 3. Precompute factorials or use math.factorial
        # We need (n-1)! to determine the first digit
        fact = math.factorial(n - 1)
        
        for i in range(n - 1, 0, -1):
            # 4. Determine the index of the digit to pick
            index = k // fact
            res.append(numbers.pop(index))
            
            # 5. Update k and calculate the factorial for the next position
            k %= fact
            fact //= i
            
        # 6. Append the last remaining number
        res.append(numbers[0])
        
        return "".join(res)
