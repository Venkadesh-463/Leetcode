class Solution:
    def spiralOrder(self, matrix: list[list[int]]) -> list[int]:
        if not matrix:
            return []
        
        res = []
        # 1. Initialize boundaries
        top, bottom = 0, len(matrix) - 1
        left, right = 0, len(matrix[0]) - 1
        
        while top <= bottom and left <= right:
            # 2. Traverse from left to right across the top row
            for i in range(left, right + 1):
                res.append(matrix[top][i])
            top += 1 # Shrink top boundary
            
            # 3. Traverse from top to bottom down the right column
            for i in range(top, bottom + 1):
                res.append(matrix[i][right])
            right -= 1 # Shrink right boundary
            
            # 4. Check if boundaries have crossed after top/right increments
            if not (top <= bottom and left <= right):
                break
            
            # 5. Traverse from right to left across the bottom row
            for i in range(right, left - 1, -1):
                res.append(matrix[bottom][i])
            bottom -= 1 # Shrink bottom boundary
            
            # 6. Traverse from bottom to top up the left column
            for i in range(bottom, top - 1, -1):
                res.append(matrix[i][left])
            left += 1 # Shrink left boundary
            
        return res
