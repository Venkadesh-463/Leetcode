class Solution:
    def totalNQueens(self, n: int) -> int:
        self.count = 0
        cols = set()
        pos_diag = set() # (r + c)
        neg_diag = set() # (r - c)
        
        def backtrack(r):
            # Base case: All queens placed successfully
            if r == n:
                self.count += 1
                return
            
            for c in range(n):
                # Check for conflicts
                if c in cols or (r + c) in pos_diag or (r - c) in neg_diag:
                    continue
                
                # 1. Place queen (Add to sets)
                cols.add(c)
                pos_diag.add(r + c)
                neg_diag.add(r - c)
                
                # 2. Move to next row
                backtrack(r + 1)
                
                # 3. Backtrack (Remove from sets)
                cols.remove(c)
                pos_diag.remove(r + c)
                neg_diag.remove(r - c)
        
        backtrack(0)
        return self.count
