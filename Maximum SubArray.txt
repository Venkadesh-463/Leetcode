class Solution:
    def maxSubArray(self, nums: list[int]) -> int:
        # Initialize current sum and global maximum with the first element
        max_sum = current_sum = nums[0]
        
        for i in range(1, len(nums)):
            # If current_sum is negative, start fresh from the current element
            # Otherwise, keep adding the current element
            current_sum = max(nums[i], current_sum + nums[i])
            
            # Update the best sum we've seen so far
            max_sum = max(max_sum, current_sum)
            
        return max_sum
