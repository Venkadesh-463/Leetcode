class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Step 1: Handle 32-bit signed integer overflow
        MAX_INT = 2**31 - 1
        MIN_INT = -2**31
        
        if dividend == MIN_INT and divisor == -1:
            return MAX_INT
        
        # Step 2: Determine the sign of the result
        negative = (dividend < 0) != (divisor < 0)
        
        # Step 3: Work with absolute values
        a, b = abs(dividend), abs(divisor)
        quotient = 0
        
        # Step 4: Use bit shifting to find the quotient
        # We try to subtract the largest possible (b << shift) from a
        for i in range(31, -1, -1):
            if (a >> i) >= b:
                a -= (b << i)
                quotient += (1 << i)
        
        # Step 5: Apply sign and return
        return -quotient if negative else quotient
