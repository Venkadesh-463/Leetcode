class Solution:
    def climbStairs(self, n: int) -> int:
        # Base cases
        if n <= 2:
            return n
        
        # first = ways to reach step 1
        # second = ways to reach step 2
        first, second = 1, 2
        
        # Calculate ways for each step from 3 to n
        for i in range(3, n + 1):
            current = first + second
            first = second
            second = current
            
        return second
