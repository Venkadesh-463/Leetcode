class Solution:
    def longestPalindrome(self, s: str) -> str:
        res = ""
        
        for i in range(len(s)):
            # Odd length palindromes (center is a character: "aba")
            pal1 = self.expand(s, i, i)
            # Even length palindromes (center is between characters: "abba")
            pal2 = self.expand(s, i, i + 1)
            
            # Keep the longest result found so far
            res = max(res, pal1, pal2, key=len)
            
        return res

    def expand(self, s, l, r):
        # Expand outwards as long as it's a palindrome
        while l >= 0 and r < len(s) and s[l] == s[r]:
            l -= 1
            r += 1
        # Return the valid substring
        return s[l + 1:r]
