class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Create a 2D grid initialized to 1
        # (The first row and first column will always be 1 because there's only one way to reach them)
        dp = [[1] * n for _ in range(m)]
        
        # Start from cell (1, 1) and fill the DP table
        for i in range(1, m):
            for j in range(1, n):
                # Paths to current cell = paths from above + paths from left
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        
        return dp[m - 1][n - 1]
