# Definition for singly-linked list.
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        # Base case: empty list, single node, or no rotation needed
        if not head or not head.next or k == 0:
            return head
        
        # 1. Calculate the length and find the last node
        old_tail = head
        length = 1
        while old_tail.next:
            old_tail = old_tail.next
            length += 1
        
        # 2. Connect the tail to the head to make it circular
        old_tail.next = head
        
        # 3. Find the new tail: (length - k % length - 1)th node
        # The new head will be at (length - k % length)th node
        k = k % length
        new_tail_steps = length - k - 1
        new_tail = head
        for _ in range(new_tail_steps):
            new_tail = new_tail.next
            
        # 4. Set the new head and break the circular connection
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
