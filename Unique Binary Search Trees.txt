class Solution:
    def numTrees(self, n: int) -> int:
        # dp[i] will store the number of unique BSTs with i nodes
        dp = [0] * (n + 1)
        
        # Base cases
        dp[0] = 1  # Empty tree is 1 way
        dp[1] = 1  # Single node is 1 way
        
        # Fill dp table for each number of nodes from 2 to n
        for nodes in range(2, n + 1):
            # Try each node 'root' as the root of the tree
            for root in range(1, nodes + 1):
                left_nodes = root - 1
                right_nodes = nodes - root
                # Total combinations for this root is (left ways * right ways)
                dp[nodes] += dp[left_nodes] * dp[right_nodes]
                
        return dp[n]
