class Solution:
    def jump(self, nums: list[int]) -> int:
        n = len(nums)
        if n <= 1:
            return 0
        
        jumps = 0
        current_end = 0
        farthest = 0
        
        # We don't need to jump from the last index
        for i in range(n - 1):
            # Update the farthest point we can reach from the current index
            farthest = max(farthest, i + nums[i])
            
            # If we've reached the end of the range for our current jump
            if i == current_end:
                jumps += 1
                current_end = farthest
                
                # Optimization: if we can already reach the end, stop
                if current_end >= n - 1:
                    break
                    
        return jumps
